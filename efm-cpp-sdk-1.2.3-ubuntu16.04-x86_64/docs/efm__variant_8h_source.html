717  bool equals(const VariantBase<size>& other) const
718  {
719  if (this->type() != other.type()) {
720  if (!((this->type() == UInt && other.type() == Int) || (this->type() == Int && other.type() == UInt))) {
721  return false;
722  }
723  }
724 
725  switch (type()) {
726  case Null:
727  return true;
728  case Bool:
729  return as_bool() == other.as_bool();
730  case Int:
731  return as_int() == other.as_int();
732  case UInt:
733  return as_uint() == other.as_uint();
734  case Double:
735  return as_double() == other.as_double();
736  case String:
737  return as_string() == other.as_string();
738  case Array:
739  return as_array() == other.as_array();
740  case Map:
741  return as_map() == other.as_map();
742  }
743  return false;
744  }

812 template <class CharT, class Traits>
813 std::basic_ostream<CharT, Traits>& operator<<(std::basic_ostream<CharT, Traits>& os, const Variant& value)
814 {
815  switch (value.type()) {
816  case Variant::Null:
817  os << "null";
818  break;
819  case Variant::Array: {
820  os << "[";
821  bool first{true};
822  for (const auto& val : value.as_array()) {
823  if (first) {
824  first = false;
825  } else {
826  os << ",";
827  }
828  os << val;
829  }
830  os << "]";
831  break;
832  }
833  case Variant::Bool:
834  os << (value.as_bool() ? "true" : "false");
835  break;
836  case Variant::Double: {
837  std::ios_base::fmtflags fmt(os.flags());
838  os << std::fixed << std::setprecision(6) << std::setfill('0') << value.as_double();
839  os.flags(fmt);
840  break;
841  }
842  case Variant::Int:
843  os << value.as_int();
844  break;
845  case Variant::UInt:
846  os << value.as_uint();
847  break;
848  case Variant::String:
849  os << "\"" << value.as_string() << "\"";
850  break;
851  case Variant::Map: {
852  os << "{";
853  bool first{true};
854  for (const auto& element : value.as_map()) {
855  if (first) {
856  first = false;
857  } else {
858  os << ",";
859  }
860  os << "\"" << element.first << "\" : " << element.second;
861  }
862  os << "}";
863  break;
864  }
865  }
